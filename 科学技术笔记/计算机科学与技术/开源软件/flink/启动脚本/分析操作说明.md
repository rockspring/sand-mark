read -a file_array <<< `ls | awk '{print $0}'`

for file in ${file_array[@]}; do
	for file_grep in ${file_array[@]}; do
		if [ "flink" = "$file_grep" ]; then
			continue
		fi
		if [ "graph.sh" = "$file_grep" ]; then
			continue
		fi
		result=`grep -E $file_grep  $file`
		if [ -z "$result" ]; then
			echo ""
		else
			echo $result
			echo $file" -> "$file_grep
		fi
	done;
done | grep "\->" | sort | uniq > a.txt

	echo $file
done
