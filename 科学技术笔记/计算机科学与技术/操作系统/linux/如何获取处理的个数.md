# 使用Linux系统命令工具


nproc - print the number of processing units available

从CPU架构信息中间接获取

lscpu - display information on CPU architecture

lscpu --parse=cpu | grep -v "^#" | wc

# 读取Linux操作系统的系统文件

grep -c processor /proc/cpuinfo
